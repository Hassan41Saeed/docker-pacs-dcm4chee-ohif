#!/bin/bash
# Create necessary groups and users for dcm4chee permissions
sudo groupadd -r slapd-dcm4chee --gid=1021 && sudo useradd -r -g slapd-dcm4chee --uid=1021 slapd-dcm4chee
sudo groupadd -r postgres-dcm4chee --gid=999 && sudo useradd -r -g postgres-dcm4chee --uid=999 postgres-dcm4chee
sudo groupadd -r dcm4chee-arc --gid=1023 && sudo useradd -r -g dcm4chee-arc --uid=1023 dcm4chee-arc

# Create working directories
sudo mkdir -p /opt/dcm4chee
sudo chown -R $USER:$USER /opt/dcm4chee
